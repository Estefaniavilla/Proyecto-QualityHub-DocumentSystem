FROM php:8.1-apache
    # Instalamos soporte para PostgreSQL
    RUN apt-get update && apt-get install -y libpq-dev \
        && docker-php-ext-install pdo pdo_pgsql pgsql
    ```

### 2. Crear el Dockerfile para Python (Módulo de Fernanda)
Este configurará FastAPI y MongoDB para las búsquedas rápidas.

1.  Ejecuta en PowerShell:
    ```powershell
    notepad docker/python/Dockerfile
    ```
2.  Pega esto en el Bloc de notas:
    
```dockerfile
    FROM python:3.10-slim
    WORKDIR /app
    # Instalamos dependencias para FastAPI y MongoDB
    RUN pip install fastapi uvicorn pymongo
    CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
    ```
3.  **Guarda** y cierra.

### 4. Crear el archivo "Maestro" (docker-compose.yml)
Este archivo es el cerebro que le dice a Docker: "Levanta estos tres Dockerfiles y conéctalos en una sola red". **Este archivo va en la carpeta principal**, no dentro de `docker`.

1.  Ejecuta en PowerShell:
    
```powershell
    notepad docker-compose.yml
    ```
2.  Pega todo este código (es el que pusiste en tu mensaje):
    
```yaml
    version: '3.8'

    services:
      # Tu Portal
      portal-php:
        build: 
          context: .
          dockerfile: docker/php/Dockerfile
        container_name: portal_php
        ports:
          - "8081:80"
        volumes:
          - ./src/php-portal:/var/www/html
          - ./storage:/var/www/html/storage:ro
        networks:
          - quality-net

      # Módulo de Fernanda
      busqueda-python:
        build:
          context: .
          dockerfile: docker/python/Dockerfile
        container_name: busqueda_python
        ports:
          - "8000:8000"
        volumes:
          - ./src/python-service:/app
          - ./storage:/app/storage:ro
        networks:
          - quality-net

      # Módulo Administrativo
      admin-dotnet:
        build:
          context: .
          dockerfile: docker/dotnet/Dockerfile
        container_name: admin_dotnet
        ports:
          - "5000:80"
        volumes:
          - ./src/dotnet-admin:/app
          - ./storage:/app/wwwroot/files
        networks:
          - quality-net

    networks:
      quality-net:
        driver: bridge
    ```
